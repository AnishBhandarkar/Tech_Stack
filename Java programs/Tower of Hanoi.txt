package com.practice;


public class Practice1{
	
	public static void toh(int n, char from, char to, char aux) {
		if(n==0)return;
		if(n==1) {
			System.out.println("Move "+n+" disks from "+from+" to "+to);
			return;
		}
		toh(n-1, from, aux, to);
		System.out.println("Move "+n+" disks from "+from+" to "+to);
		toh(n-1, aux, to, from);
	}
	
	
	public static void main(String[] args) {
		toh(10, 'A', 'B', 'C');
	}
}
