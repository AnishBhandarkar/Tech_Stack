T(n) = O(n)
S(n) = O(n)


public class Practice1 {

	public static void main(String[] args) {
		int n =9; //upperbound
		int[] arr = {2,3,7,8,9};
		HashMap<Integer, Integer> map = new HashMap<>();
		for(int i=1;i<=n;i++) {
			map.put(i, 0);
		}
		for(int i=0;i<arr.length;i++) {
			map.put(arr[i], 1);
		}
		System.out.println("Missing elements are :");
		for(int i=1;i<=n;i++) {
			if(map.get(i) == 0) {
				System.out.print(i+" ");
			}
		}
		
	}
}
